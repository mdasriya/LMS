                                  setTransferAllPlotName(e.target.value);

                                    </option>
                                  );

                              </Select>
                            </Box>
                          </Flex>
                        </FormControl>
                      </ModalBody>

                      <ModalFooter>
                        <Button
                          colorScheme="blue"
                          mr={3}
                          onClick={confirmTransferAll}

                          Transfer Transaction
                        </Button>
                        <Button onClick={() => setIsTransferModalOpen(false)}>
                          Cancel
                        </Button>
                      </ModalFooter>
                    </ModalContent>
                  </Modal>
                </Tbody>
              </Table>
            </TableContainer>
            {showAction && (
              <>
                <Button onClick={handlePrint} className="hide-on-print">
                  Print
                </Button>
              </>
            )}
          </Box>
        </Box>
      </Box>
    </Box>
  );
};
